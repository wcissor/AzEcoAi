1    import os
2    import joblib
3    import pandas as pd
4    import csv
5    from datetime import datetime
6    from flask import Flask,request,jsonify,render_template
7    from config import SERVER_PORT,ICE_RISK_THRESHOLDS
8    app = Flask(__name__)
9    LOG_FILE = "sensor_log.csv"
10   if not os.path.exists(LOG_FILE):
11       with open(LOG_FILE,"w",newline = "") as f:
12           writer = csv.writer(f)
13           writer.writerow(["temperature","humidity","timestamp","road_moisture"])
14   
15   ice_model = joblib.load("models/ice_model.pkl")
16   features = joblib.load("models/features.pkl")
17   def create_ice_features(data):
18       temperature = data["temperature"]
19       road_moisture= data["road_moisture"]
20       humidity = data["humidity"]
21       freezing_index = max(0,0 - temperature)* road_moisture
22       dew_point = temperature - ((100 - humidity)/5)
23       features_data = {"temperature":temperature,
24                        "humidity":humidity,
25                        "freezing_index":freezing_index,
26                        "dew_point":dew_point,
27                        "road_moisture":road_moisture}
28       return pd.DataFrame([features_data])[features]
29   @app.route("/ice_predict",methods = ["POST"])
30   def predict_ice():
31       data = request.json
32       required_fields = ["temperature","humidity","road_moisture"]
33       for field in required_fields:
34           if field not in required_fields:
35               return jsonify({"error",f"{field}:missing"}),400
36       try:
37           temperature = float(data["temperature"])
38           humidity = float(data["humidity"])
39           road_moisture = float(data["road_moisture"])
40       except:
41           return jsonify({"error","NZD"})
42   
43       with open(LOG_FILE,"a",newline="") as f:
44           writer = csv.writer(f)
45           writer.writerow([datetime.now().strftime("%H:%M:%S"),road_moisture,temperature,humidity])
46       df = create_ice_features({"temperature":temperature,"humidity":humidity,"road_moisture":road_moisture})
47       probability = ice_model.pred_proba(df)[0][1]
48       if probability > ICE_RISK_THRESHOLDS["HIGH"]:
49           risk = "HIGH"
50       elif probability > ICE_RISK_THRESHOLDS["LOW"]:
51           risk = "MEDIUM"
52       else:
53           risk = "LOW"
54       return jsonify({"ice_probability":float(probability),
55                       "ice_risk":risk})
56   @app.route("/sensor_log")
57   def sensor_log():
58       if not os.path.exists(LOG_FILE):
59   
60           return jsonify([])
61           df = pd.read_csv(LOG_FILE)
62           df = df.tail(30)
63           return df.to_json(orient = "records")
