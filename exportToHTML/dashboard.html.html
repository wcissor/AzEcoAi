1259 function computePrediction(temp, hum, moist) {
1260   const dewPoint = temp - ((100 - hum) / 5);
1261   const freezeIdx = Math.max(0, -temp) * moist;
1262   let prob = 0;
1263   if (temp < 0) prob += 0.35 * Math.min(1, Math.abs(temp) / 5);
1264   if (dewPoint < 0) prob += 0.25 * Math.min(1, Math.abs(dewPoint) / 5);
1265   prob += 0.2 * moist;
1266   prob += 0.15 * Math.min(1, hum / 100);
1267   if (freezeIdx > 0) prob += 0.05 * Math.min(1, freezeIdx / 3);
1268   prob = Math.max(0, Math.min(0.99, prob));
1269   return { prob, dewPoint, freezeIdx };
1270 }
