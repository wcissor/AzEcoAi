1    import os
2    import joblib
3    import pandas as pd
4    import numpy as np
5    from sklearn.ensemble import RandomForestClassifier
6    from sklearn.model_selection import train_test_split
7    from sklearn.metrics import accuracy_score,classification_report
8    DATA_PATH = "data/ice_data.csv"
9    MODEL_PATH = "models/ice_model.pkl"
10   FEATURES_PATH = "models/features.pkl"
11   if not os.path.exists(DATA_PATH):
12       raise FileNotFoundError(f"oops:{DATA_PATH}")
13   df = pd.read_csv(DATA_PATH)
14   print(df.shape)
15   df = df.dropna()
16   df =df[df["humidity"]<= 100]
17   df =df[df["road_moisture"]>= 0]
18   def add_features(data):
19       data["dew_point"] = data["temperature"] - ((100 - data["humidity"])/5)
20       data["freezing_index"] = np.maximum(0,0 - data["temperature"]) * data["road_moisture"]
21       return data
22   df = add_features(df)
23   FEATURES = ["dew_point","freezing_index","temperature","humidity","road_moisture"]
24   TARGET =  "ice_label"
25   X = df[FEATURES]
26   y = df[TARGET]
27   X_train,X_test,y_train,y_test = train_test_split(X,y, test_size = 0.3,random_state = 42)
28   print(len(X_train))
29   print(len(X_test))
30   model = RandomForestClassifier(n_estimators=200,max_depth=10,min_samples_split=5,random_state=42)
31   model.fit(X_train,y_train)
32   y_pred = model.predict(X_test)
33   accuracy = accuracy_score(y_test,y_pred)
34   print(f"classification report:", classification_report(y_test,y_pred))
35   for feature,importance in zip(FEATURES,model.feature_importances_):
36       print(f"{feature}: {importance:.3f}")
37   
38   os.makedirs("models",exist_ok = True)
39   joblib.dump(model,MODEL_PATH)
40   joblib.dump(FEATURES,FEATURES_PATH)
41   sample = {"temperature":-1.5,"humidity":85,"road_moisture":0.8}
42   sample["dew_point"] = sample["temperature"] - ((100 - sample["humidity"])/5)
43   sample["freezing_index"] = max(0,0 - sample["temperature"])* sample["road_moisture"]
44   sample_df = pd.DataFrame([sample])[FEATURES]
45   probability = model.predict_proba(sample_df)[0][1]
